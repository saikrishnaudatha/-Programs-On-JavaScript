            var n = 1;

                sum += n;
                n++;
            }
            document.write("Sum is : " + sum);

            //do-while loop
            do {
            sum += n;
                n++;
            } while (n < 10);
            document.write("Sum is : " + sum);


            //for loop

            sum = 0;
            for (var i = 0; i < 10; i++) {
                sum += n;
            }
             document.write("Sum is : " + sum);
        }
